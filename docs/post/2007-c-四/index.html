#define N 10

typedef struct LNode {
int data;
struct LNode *next;
} LNode;

//建立线性表
LNode* create()
{
LNode *c,*s,*r;
int i;
c=(LNode*)malloc(sizeof(LNode));
c->next=NULL;
r=c;
for(i=1;i<=N;++i) { s=(LNode*)malloc(sizeof(LNode)); s->data=i;
s->next=NULL;
r->next=s;
r=r->next;
}
return c;
}
//删除线性表某个节点
void seaDel(LNode *c,int x)
{
LNode *p,*q;
p=c;
//查找
while(p->next!=NULL)
{
if(p->next->data==x)
break;
p=p->next;
}
//删除
if(p->next!=NULL)
{
q=p->next;
p->next=p->next->next;
free(q);
}
}
//输出
void out(LNode *c)
{
LNode *p;
p=c->next;
while(p!=NULL)
{
printf("%d-->",p->data);
p=p->next;
}
printf("\n");
}
