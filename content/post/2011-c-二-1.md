#include <stdio.h>
#include <stdlib.h>
int main()
{
    //定义文件指针
    FILE *p;
    char c;
    int num=0;
    //打开文件
    if((p=fopen(".\\demo.txt","r"))==NULL)
    {
        printf("error");
        exit(-1);
    }
    while(fscanf(p,"%c",&c)!=EOF)
    {
        ++num;
        printf("%c",c);
        if(num==10)
        {
            printf("\n");
            num=0;
        }

    }
}
