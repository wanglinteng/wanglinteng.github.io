int similar(BTNode *n1,BTNode *n2)
{
    if((n1==NULL)&&(n2==NULL))//都为空
        return 1;//相似
    else//有一个不空，或都不空
    {
        if((n1==NULL)||(n2==NULL))//有一个不空
            return 0;
        else
            return similar(n1->lchild,n2->lchild)&&similar(n1->rchild,n2->rchild);//继续判断
    }
}
