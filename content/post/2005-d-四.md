int visit[maxSize];//初始化全为0
void DFS(AGraph *G,int v,int t)
{
    ArcNode *p;
    visit[v]=1;
    p=G->adjlist[v].firstarc;
    while(p!=NULL)
    {
        //在深度优先搜索中添加部分
        if(p->adjvex==t)
        {
            printf("exist");
            break;
        }else{
            if(visit[p->adjvex]==0)
            {
                DFS(G,p->adjvex,t);
                p=p->nextarc;
            }
        }
    }
}
