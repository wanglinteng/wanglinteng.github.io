```cpp

//递增：1，不递增：0
int dec(int a[],int n)
{  
    if(n==0)
        return 1;  //递增

    if(a[n]>a[n-1])
         return dec(a,n-1);
    else
        return 0;  //不递增
}

```

int dec(int a[], int n)
{
    //printf("n=%d,",n);
    //printf("a[0]=%d\n",a[0]);
  if (n <= 1)
   {
        return 0;//非递增
   }
   if (a[0] < a[1])
   {
        return dec(++a,--n);//注意n--是先把值给dec再减
   }else{
        return 1;//递增
   }

}

int main()
{
   int a[]={1,2,9,5};
   printf("%d",dec(a,4));
}
