/**
    基本思想：采取逆先序遍历方法，判断结点是否为叶子节点，在访问右子树、最后访问左子树
**/
int arr[maxSize];int k=0;
void del(BTNode *bt)
{
    if(bt!=NULL)
    {
      if(bt->lchld==NULL&&bt->rchild==NULL)
      {
        arr[k++]=bt->data;
        free(bt);
        bt=NULL;
      }
      else
      {
          del(bt->rchild);//注意此时的顺序
          del(bt->lchild);
      }
    }
    
}
