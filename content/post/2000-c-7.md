void merge(int arg1[],int n,int arg2[],int m)
{
    int arg3[maxSize];
    int i=0,j=0,k=0,l=0;
    while(i<n&&j<m)
    {
        if(arg1[i]<arg2[j]) arg3[k++]=arg1[i++];
        if(arg1[i]>arg2[j]) arg3[k++]=arg2[j++];
    }
    if(i<n) 
        while(i<n) arg3[k++]=arg1[i++];
    if(j<m)
        while(j<m) arg3[k++]=arg2[j++];
    while(l<k)//复制到arg1中
    {
        arg1[l]=arg3[l];
        ++l;
    }
}
