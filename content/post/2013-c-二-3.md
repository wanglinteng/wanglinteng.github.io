#include <stdio.h>
#include <stdlib.h>

int main()
{
    char str[50][32];//用于存储从文件中读取的单词
    char temp[32];//用于临时存储变量
    FILE *in,*out;
    int i,j,n=0;
    //判断打开文件是否成功
    if((in=fopen(".\\file.txt","r"))==NULL)//文件当前路径
    {
        printf("open error");
        exit(-1);//需要 stdlib.h
    }
    //读取文件
    while(fscanf(in,"%s ",str[n])!=EOF) n++;
    //关闭文件
    fclose(in);

    //对单词进行排序，采用冒泡法
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
        {
            if(strcmp(str[j],str[j+1])>0)
            {
                strcpy(temp,str[j+1]);
                strcpy(str[j+1],str[j]);
                strcpy(str[j],temp);
            }
        }
    }
    //输出文件

    //判断打开文件是否成功
    if((out=fopen(".\\file2.txt","w"))==NULL)
    {
        printf("open error");
        exit(-1);//需要 stdlib.h
    }
    for(i=0;i<n;i++) fprintf(out,"%s ",str[i]);
    fclose(out);
}
