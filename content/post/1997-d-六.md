// 先序遍历树中每个节点 
int visit( int T[],int n,int id)
{
	if(id > n) return 0;
	printf("%d",id);
	visit(T,n,id*2);
	visit(T,n,id*2+1);
} 

//打印全部祖先结点
int print_ancestor(int T[],int n)
{
	int parent = 0;
	parent = n/2;
	while(parent >0)
	{
		printf("%d",parent);
		parent = parent/2;
	}
 }
