```cpp
#include <stdio.h>

int main()
{
    int k;
    char c;
    for(k=1,c='A';c<'F';k++)
    {
        switch(++c)
        {
            case 'A':k++;break;
            case 'B':k*=2;break;
            case 'C':k--;
            case 'D':k%=3;continue;
            default:k+=2;
            case 'E':k/=2;
            case 'F':k++;
        }
        k++;
    }
    printf("%d",k);
}
```
