void swap(BTNode *T)
{
    BTNode *p,*temp;
    p=T;
    if(p!=NULL)
    {
        //交换左右子树
        temp=p->rchild;
        p->rchild=p->lchild;
        p->lchild=temp;
        //递归实现全部指针互换
        swap(p->lchild);
        swap(p->rchild);
    }
}
