#include <stdio.h>

#define maxSize 100

void delstring(char *pstr1,char *pstr2)
{
    char *p,*q,*t;
    int len=0;//子串长度
    p=pstr1;
    q=pstr2;
    while(*q!='\0')
    {
      ++len;//求子串长度
      ++q;
    }

    while(*p!='\0')
    {
        q=pstr2;//每次完成后子串指针复位
        while(*p==*q&&*q!='\0')//匹配子串
        {
            p++;
            q++;
        }
        if(*q=='\0')
        {
            t=p-len;
            while(*p!='\0')
            {
                *(p-len)=*p;
                p++;
            }
            *(p-len)='\0';
            p=t;
        }else{
            ++p;
        }
    }
    printf("%s",pstr1);
}
int main()
{
    char a[maxSize],b[maxSize];
    scanf("%s",a);
    scanf("%s",b);
    printf("%s\n",a);
    delstring(a,b);
}
