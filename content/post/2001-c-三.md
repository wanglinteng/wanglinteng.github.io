void common(char *str,char *sub)
{
    int i=0,j=0,k=0,max=0;
    for(i=0;str[i]!='\0';++i)
    {
        j=0;
        k=i;
        while(str[k]==sub[j])
        {
            ++k;
            ++j;
        }
        if(j>max) max=j;
    }

    sub[max]='\0';
    printf("%s",sub);
}

int main()
{
    char str1[40]="123456123456vhgaaasdd";
    char str2[20]="123456vhgvhg";
    common(str1,str2);
}
