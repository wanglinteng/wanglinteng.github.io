```cpp
#include <stdio.h>
//功能函数
int total(int n)
{
    if(n==1) return 1;
    else{
        return total(n-1)+n+1;
    }
}
//主函数
int main()
{
    int n;
    scanf("%d",&n);
    printf("%d",total(n));
}
```
